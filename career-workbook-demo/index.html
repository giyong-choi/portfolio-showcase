JWT Custom Claims 구조Supabase Auth Hook
-- 로그인 시 JWT에 role + school_id 자동 삽입 CREATE OR REPLACE FUNCTION custom_access_token_hook(event JSONB) RETURNS JSONB LANGUAGE plpgsql AS $$ BEGIN claims := jsonb_set(claims, '{user_role}', to_jsonb(user_role)); claims := jsonb_set(claims, '{school_id}', to_jsonb(school_id)); RETURN event; END; $$;
